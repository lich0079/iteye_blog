function simpleQuery() {
var time = Ext.get('datetype').dom.value;
var timeb = time + 'b';
var timee = time + 'e';
store.removeAll();
store.load({
params : {
timeb : Ext.get('stimeb').dom.value,
timee : Ext.get('stimee').dom.value,
start : 0,
limit : 10,
},
callback : function(r, options, success) {
if (!success) {
Ext.Msg.alert("信息提示", " 连接服务器出现错误,请稍后再试！");
}

}
});
store.baseParams = {
timeb : Ext.get('stimeb').dom.value,
timee : Ext.get('stimee').dom.value

}
}

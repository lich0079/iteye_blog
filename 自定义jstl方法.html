/**
	 * 自定义函数库实现
	 * @author Administrator
	 *
	 */
	public class MyFunctions {
		
		/**
		 * 方法必须是public静态的
		 * @param name
		 * @return
		 */
		public static String sayHello(String name) {
			return "Hello " + name;
		}
	}
